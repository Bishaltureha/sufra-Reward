// Helper functions for working with countries data
export const getCountryByCode = (code) => {
  return countries.find((country) => country.code === code);
};

export const getCountryByDialCode = (dialCode) => {
  return countries.find((country) => country.dial_code === dialCode);
};

export const getCountriesByName = (searchTerm) => {
  return countries.filter((country) =>
    country.name.toLowerCase().includes(searchTerm.toLowerCase())
  );
};

export const validatePhoneNumber = (countryCode, phoneNumber) => {
  const country = getCountryByCode(countryCode);
  if (!country) return false;

  // Remove any non-digit characters
  const cleanNumber = phoneNumber.replace(/\D/g, "");

  // Check if the length matches expected phone length
  return cleanNumber.length === country.phoneLength;
};

export const formatPhoneNumber = (countryCode, phoneNumber) => {
  const country = getCountryByCode(countryCode);
  if (!country) return phoneNumber;

  const cleanNumber = phoneNumber.replace(/\D/g, "");
  return `${country.dial_code} ${cleanNumber}`;
};
